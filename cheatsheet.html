Basic Template

import { chromium } from "playwright";

async function scrape() {
  const browser = await chromium.launch({ headless: false });
  const page = await browser.newPage();
  
  await page.goto("YOUR-URL-HERE");
  
  // Your scraping code here
  
  await browser.close();
}

scrape();

Navigation

Go to Pages

// Go to a page
await page.goto("https://example.com");

// Go to a page with custom timeout
await page.goto("https://example.com", { timeout: 60000 });

// Go back/forward
await page.goBack();
await page.goForward();

// Reload
await page.reload();

Interact with Elements

// Click something
await page.click("button.submit");

// Double click
await page.dblclick("button");

// Fill in a form
await page.fill("input#email", "test@example.com");

// Type text (slower, more realistic)
await page.type("input", "Hello World");

// Press a key
await page.press("input", "Enter");

// Check a checkbox
await page.check("input[type='checkbox']");

// Select from dropdown
await page.selectOption("select", "value");

Finding Elements (CSS Selectors)

What You Want Selector Example HTML
By class .classname <div class="classname">
By ID #idname <div id="idname">
By tag tagname <h1>, <p>, <a>
Multiple classes .class1.class2 <div class="class1 class2">
Child element .parent .child Any descendant
Direct child .parent > .child Immediate child only
By attribute [href] <a href="...">
Specific attribute [href="/about"] Exact match
Right-click any element on a webpage and choose "Inspect" to see its HTML and find the right selector!

Getting Data

Get Text from Elements

// Get text from ONE element
const text = await page.textContent("h1");

// Get text from MANY elements
const items = await page.$$eval("li", els => 
  els.map(el => el.textContent)
);

// Get inner HTML
const html = await page.innerHTML(".content");

Get Attributes

// Get an attribute (like href, src, class, etc.)
const url = await page.getAttribute("a", "href");
const imgSrc = await page.getAttribute("img", "src");

// Get multiple attributes
const links = await page.$$eval("a", anchors =>
  anchors.map(a => ({
    href: a.href,
    text: a.textContent
  }))
);

Complex Data Extraction

// Get multiple pieces of data
const products = await page.$$eval(".product", items => {
  return items.map(item => ({
    name: item.querySelector(".name")?.textContent,
    price: item.querySelector(".price")?.textContent,
    image: item.querySelector("img")?.src,
  }));
});

// Check if element exists
const exists = await page.locator(".element").count() > 0;

// Get current URL
const currentUrl = await page.url();

// Get page title
const title = await page.title();

Waiting for Things

// Wait for an element to appear
await page.waitForSelector(".results");

// Wait for page to load
await page.waitForLoadState("domcontentloaded");
await page.waitForLoadState("networkidle"); // wait for all network requests

// Wait a specific time (milliseconds)
await page.waitForTimeout(3000); // 3 seconds

// Wait for navigation
await page.waitForNavigation();

// Wait for element to be visible
await page.waitForSelector(".element", { state: "visible" });

// Wait for element to be hidden
await page.waitForSelector(".element", { state: "hidden" });
Use timeouts sparingly! It's better to wait for specific elements than arbitrary time periods.

Browser Options

// Visible browser (good for learning)
const browser = await chromium.launch({ 
  headless: false 
});

// Invisible browser (faster)
const browser = await chromium.launch({ 
  headless: true 
});

// Slow down actions (easier to watch)
const browser = await chromium.launch({ 
  headless: false,
  slowMo: 1000 // 1 second delay between actions
});

// Custom window size
const page = await browser.newPage();
await page.setViewportSize({ width: 1920, height: 1080 });

Common Patterns

Pattern 1: Get All Links

const links = await page.$$eval("a", anchors => 
  anchors.map(a => ({
    text: a.textContent,
    url: a.href
  }))
);

Pattern 2: Get All Images

const images = await page.$$eval("img", imgs => 
  imgs.map(img => ({
    src: img.src,
    alt: img.alt
  }))
);

Pattern 3: Login to a Site

await page.goto("https://example.com/login");
await page.fill("#username", "myusername");
await page.fill("#password", "mypassword");
await page.click("button[type='submit']");
await page.waitForNavigation();

Pattern 4: Scroll to Load More

// Scroll to bottom
await page.evaluate(() => 
  window.scrollTo(0, document.body.scrollHeight)
);

// Wait for new content
await page.waitForTimeout(2000);

Pattern 5: Handle Popups

// Click something that opens a new tab
const [newPage] = await Promise.all([
  page.waitForEvent('popup'),
  page.click('a[target="_blank"]')
]);

// Work with the new page
await newPage.waitForLoadState();
const text = await newPage.textContent('h1');

Error Handling

try {
  await page.goto("https://example.com");
  const data = await page.textContent(".selector");
  console.log(data);
} catch (error) {
  console.error("Something went wrong:", error.message);
  
  if (error.message.includes("timeout")) {
    console.log("Page took too long to load");
  } else if (error.message.includes("Target closed")) {
    console.log("Browser was closed");
  }
} finally {
  await browser.close();
}

Debugging Tips

// Pause execution (opens inspector)
await page.pause();

// Print current URL
console.log(await page.url());

// Print page title
console.log(await page.title());

// Get the full HTML
const html = await page.content();
console.log(html);

// Screenshot when something goes wrong
try {
  // ... your code
} catch (error) {
  await page.screenshot({ path: "error.png" });
  throw error;
}

Tips & Best Practices

DO:

  • Use headless: false while learning
  • Add waits for dynamic content
  • Check if elements exist before accessing
  • Use try/catch for error handling
  • Be respectful - add delays between requests
  • Test selectors in browser DevTools first

DON'T:

  • Scrape too fast (can get you blocked)
  • Ignore robots.txt
  • Hardcode credentials in scripts
  • Forget to close the browser
  • Assume elements will always be there
  • Use excessive waitForTimeout()
